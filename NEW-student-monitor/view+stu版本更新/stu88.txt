def is_valid_public_url(url):
    if not url:
        return False
    url = url.strip().lower()
    # æ’é™¤å†…éƒ¨åè®®ã€æ— æ•ˆåœ°å€
    invalid_prefixes = (
        'about:', 'chrome:', 'edge:', 'file:', 'moz-extension:',
        'javascript:', '', 'ftp:', 'mailto:', 'blob:'
    )
    if any(url.startswith(p) for p in invalid_prefixes):
        return False
    # ä»…å…è®¸ http/https
    return url.startswith(('http://', 'https://'))
